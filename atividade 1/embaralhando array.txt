class Solution {
public:
    vector<int> shuffle(vector<int>& nums, int n) {
        vector<int> V;
        
        int i = 0;
        int j = n;
        
        while (i< n && j< 2*n){
            V.push_back(nums[i]);
            V.push_back(nums[j]);
            i++;
            j++;
        }
        return V;
    }
};
